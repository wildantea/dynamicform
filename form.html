$(document).ready(function() { var max_fields_limit = 10; //set limit for maximum input fields $(document).on('click', '.btn-add-peserta', function(e) { e.preventDefault(); // var x = 1; //initialize counter for text box var x = $(".clone_peserta").length+1; if(x <= max_fields_limit){ //check conditions var controlForm = $('.cloneable_pesertas:first'), currentEntry = $(this).parents('.clone_peserta:first'), newEntry = $(currentEntry.clone()).appendTo(controlForm); newEntry.find('.numbering').text(x); controlForm.find('.clone_peserta:not(:last) .btn-add-peserta') .removeClass('btn-add-peserta').addClass('btn-remove-peserta') .removeClass('btn-success').addClass('btn-danger') .html(''); x++; //counter increment } }); $(document).on('click', '.btn-remove-peserta', function(e) { $(this).parents('.clone_peserta:first').remove(); var new_number = 1; $('.numbering').each(function() { $(this).text(new_number) new_number++; }) e.preventDefault(); return false; }); });
